--Integration DB

CREATE PROCEDURE STG.truncate_staging_tables

AS

BEGIN

    SET NOCOUNT ON;



    TRUNCATE TABLE STG_COUNTRY;

    TRUNCATE TABLE STG_ORDER_METHOD;

    TRUNCATE TABLE STG_PRODUCT;

    TRUNCATE TABLE STG_PRODUCT_NAME_LOOKUP;

    TRUNCATE TABLE STG_RETAILER;

    TRUNCATE TABLE STG_RETURN_REASON;

    TRUNCATE TABLE STG_WAREHOUSE;



    PRINT 'Staging tables truncated successfully';

END;
