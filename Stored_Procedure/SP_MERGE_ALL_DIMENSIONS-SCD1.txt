CREATE PROCEDURE STG.SP_MERGE_ALL_DIMENSIONS	
  AS	
  BEGIN	

      SET NOCOUNT ON;	
      -- =====================================================	
      -- 1. ORDER_METHOD	
      -- =====================================================	
      MERGE dbo.TBL_DIM_ORDER_METHOD_LKP AS tgt	
      USING STG.STG_ORDER_METHOD AS src	
      ON tgt.METHOD_CODE = src.METHOD_CODE	
      WHEN MATCHED THEN	

      UPDATE SET	

          tgt.METHOD_NAME = src.METHOD_NAME,	

          tgt.SOURCE_ID = src.SOURCE_ID,	

          tgt.DATA_DATE = src.DATA_DATE,	

          tgt.UPDATE_DATE = GETDATE()


      WHEN NOT MATCHED THEN	

      INSERT	

      (	


          METHOD_CODE,	


          METHOD_NAME,	


          SOURCE_ID,	


          DATA_DATE,	


          UPDATE_DATE	


      )	


      VALUES	


      (	


          src.METHOD_CODE,	


          src.METHOD_NAME,	


          src.SOURCE_ID,	


          src.DATA_DATE,	


          GETDATE()	


      );	


  	


      -- =====================================================	


      -- 2. RETURN_REASON	


      -- =====================================================	


  	


      MERGE dbo.TBL_DIM_RETURN_REASON_LKP AS tgt	


      USING STG.STG_RETURN_REASON AS src	


      ON tgt.RETURN_REASON_CODE = src.RETURN_REASON_CODE	


  	


      WHEN MATCHED THEN	


      UPDATE SET	


          tgt.RETURN_REASON_DESC = src.RETURN_REASON_DESC,	


          tgt.SOURCE_ID = src.SOURCE_ID,	


          tgt.DATA_DATE = src.DATA_DATE,	


          tgt.UPDATE_DATE = GETDATE()	


  	


      WHEN NOT MATCHED THEN	


      INSERT	


      (	


          RETURN_REASON_CODE,	


          RETURN_REASON_DESC,	


          SOURCE_ID,	


          DATA_DATE,	


          UPDATE_DATE	


      )	


      VALUES	


      (	


          src.RETURN_REASON_CODE,	


          src.RETURN_REASON_DESC,	


          src.SOURCE_ID,	


          src.DATA_DATE,	


          GETDATE()	


      );	


  	


      -- =====================================================	


      -- 3. COUNTRY	


      -- =====================================================	


  	


      MERGE dbo.TBL_DIM_COUNTRY_LKP AS tgt	


      USING STG.STG_COUNTRY AS src	


      ON tgt.COUNTRY_CODE = src.COUNTRY_CODE	


  	


      WHEN MATCHED THEN	


      UPDATE SET	


          tgt.COUNTRY_ID = src.COUNTRY_ID,	


          tgt.COUNTRY_NAME = src.COUNTRY_NAME,	


          tgt.SOURCE_ID = src.SOURCE_ID,	


          tgt.DATA_DATE = src.DATA_DATE,	


          tgt.UPDATE_DATE = GETDATE()	


  	


      WHEN NOT MATCHED THEN	


      INSERT	


      (	


          COUNTRY_CODE,	


          COUNTRY_ID,	


          COUNTRY_NAME,	


          SOURCE_ID,	


          DATA_DATE,	


          UPDATE_DATE	


      )	


      VALUES	


      (	


          src.COUNTRY_CODE,	


          src.COUNTRY_ID,	


          src.COUNTRY_NAME,	


          src.SOURCE_ID,	


          src.DATA_DATE,	


          GETDATE()	


      );	


  	


      -- =====================================================	


      -- 4. WAREHOUSE	


      -- =====================================================	


  	


      MERGE dbo.TBL_DIM_WAREHOUSE_LKP AS tgt	


      USING STG.STG_WAREHOUSE AS src	


      ON tgt.BRANCH_CODE = src.BRANCH_CODE	


  	


      WHEN MATCHED THEN	


      UPDATE SET	


          tgt.ADDRESS = src.ADDRESS,	


          tgt.CITY = src.CITY,	


          tgt.POSTAL_CODE = src.POSTAL_CODE,	


          tgt.COUNTRY_CODE = src.COUNTRY_CODE,	


          tgt.WAREHOUSE_BRANCH_CODE = src.WAREHOUSE_BRANCH_CODE,	


          tgt.SOURCE_ID = src.SOURCE_ID,	


          tgt.DATA_DATE = src.DATA_DATE,	


          tgt.UPDATE_DATE = GETDATE()	


  	


      WHEN NOT MATCHED THEN	


      INSERT	


      (	


          BRANCH_CODE,	


          ADDRESS,	


          CITY,	


          POSTAL_CODE,	


          COUNTRY_CODE,	


          WAREHOUSE_BRANCH_CODE,	


          SOURCE_ID,	


          DATA_DATE,	


          UPDATE_DATE	


      )	


      VALUES	


      (	


          src.BRANCH_CODE,	


          src.ADDRESS,	


          src.CITY,	


          src.POSTAL_CODE,	


          src.COUNTRY_CODE,	


          src.WAREHOUSE_BRANCH_CODE,	


          src.SOURCE_ID,	


          src.DATA_DATE,	


          GETDATE()	


      );	


  	


      -- =====================================================	


      -- 5. RETAILER	


      -- =====================================================	


  	


      MERGE dbo.TBL_DIM_RETAILER_LKP AS tgt	


      USING STG.STG_RETAILER AS src	


      ON tgt.RETAILER_SITE_CODE = src.RETAILER_SITE_CODE	


  	


      WHEN MATCHED THEN	


      UPDATE SET	


          tgt.RETAILER_NAME = src.RETAILER_NAME,	


          tgt.RETAILER_CONTACT_CODE = src.RETAILER_CONTACT_CODE,	


          tgt.SOURCE_ID = src.SOURCE_ID,	


          tgt.DATA_DATE = src.DATA_DATE,	


          tgt.UPDATE_DATE = GETDATE()	


  	


      WHEN NOT MATCHED THEN	


      INSERT	


      (	


          RETAILER_NAME,	


          RETAILER_SITE_CODE,	


          RETAILER_CONTACT_CODE,	


          SOURCE_ID,	


          DATA_DATE,	


          UPDATE_DATE	


      )	


      VALUES	


      (	


          src.RETAILER_NAME,	


          src.RETAILER_SITE_CODE,	


          src.RETAILER_CONTACT_CODE,	


          src.SOURCE_ID,	


          src.DATA_DATE,	


          GETDATE()	


      );	


  	


      -- =====================================================	


      -- 6. PRODUCT_NAME	


      -- =====================================================	


  	


      MERGE dbo.TBL_DIM_PRODUCT_NAME_LKP AS tgt	


      USING STG.STG_PRODUCT_NAME_LOOKUP AS src	


      ON tgt.PRODUCT_NUMBER = src.PRODUCT_NUMBER	


  	


      WHEN MATCHED THEN	


      UPDATE SET	


          tgt.PRODUCT_NAME = src.PRODUCT_NAME,	


          tgt.PRODUCT_DESCRIPTION = src.PRODUCT_DESCRIPTION,	


          tgt.SOURCE_ID = src.SOURCE_ID,	


          tgt.DATA_DATE = src.DATA_DATE,	


          tgt.UPDATE_DATE = GETDATE()	


  	


      WHEN NOT MATCHED THEN	


      INSERT	


      (	


          PRODUCT_NUMBER,	


          PRODUCT_NAME,	


          PRODUCT_DESCRIPTION,	


          SOURCE_ID,	


          DATA_DATE,	


          UPDATE_DATE	


      )	


      VALUES	


      (	


          src.PRODUCT_NUMBER,	


          src.PRODUCT_NAME,	


          src.PRODUCT_DESCRIPTION,	


          src.SOURCE_ID,	


          src.DATA_DATE,	


          GETDATE()	


      );	


  	


  END
